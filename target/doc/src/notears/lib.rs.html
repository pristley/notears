1//! # NOTEARS: Nonlinear ICA for Structure Learning
2//!
3//! A production-grade Rust implementation of the NO TEARS algorithm for learning
4//! directed acyclic graph (DAG) structures from observational data.
5//!
6//! ## Paper
7//! Zheng, X., Aragam, B., Ravikumar, P. K., & Xing, E. P. (2018).
8//! "DAGs with NO TEARS: Continuous optimization for learning acyclic graphs."
9//! Advances in Neural Information Processing Systems (NeurIPS).
10//! https://arxiv.org/abs/1803.01422
11//!
12//! ## Algorithm Overview
13//!
14//! The NO TEARS algorithm solves:
15//! ```text
16//! minimize F(W) = (1/2n)||X - XW||_F^2 + λ||W||_1
17//! subject to h(W) = tr(exp(W ∘ W^T)) - d = 0
18//! ```
19//!
20//! where:
21//! - `W` is the d×d weight matrix defining a DAG
22//! - `h(W)` is the acyclicity constraint (equals 0 iff W is acyclic)
23//! - `∘` denotes element-wise Hadamard product
24//! - `exp` is the matrix exponential
25//!
26//! ## Key Features
27//!
28//! - **O(d³) per-iteration complexity** via matrix exponential computation
29//! - **Differentiable acyclicity constraint** enabling gradient-based optimization
30//! - **L-BFGS + Augmented Lagrangian** for constrained optimization
31//! - **Numerical stability** across varying data regimes
32//! - **Production-grade error handling** and validation
33//!
34//! ## Quick Start
35//!
36//! ```rust,no_run
37//! use notears::optimization::solve;
38//! use notears::utils::standardize_data;
39//! use ndarray::Array2;
40//!
41//! // Prepare data (n samples, d variables)
42//! let data = Array2::zeros((100, 5));
43//! let standardized = standardize_data(&data)?;
44//!
45//! // Learn DAG structure
46//! let result = solve(&standardized, 0.1)?; // lambda = 0.1
47//!
48//! // Extract edges (detected via edge_threshold)
49//! let edges = result.edges();
50//! println!("Learned edges: {:?}", edges);
51//! # Ok::<(), Box<dyn std::error::Error>>(())
52//! ```
53//!
54//! ## Module Organization
55//!
56//! - `types` - Core data structures and configuration
57//! - `acyclicity` - h(W) constraint and gradient computation
58//! - `scoring` - Loss functions (MSE, L1, L2)
59//! - `optimization` - L-BFGS + Augmented Lagrangian solver
60//! - `utils` - Matrix operations, validation, standardization
61

85/// High-level API for learning directed acyclic graph structure from data
86///
87/// This is the main entry point for users of the NOTEARS library. It orchestrates
88/// the complete pipeline: data validation, weight initialization, optimization,
89/// and post-processing to extract discrete DAG structure.
90///
91/// # Arguments
92///
93/// * `data` - Observation matrix (n × d) where n = # samples, d = # variables
94/// * `lambda` - L₁ regularization coefficient, typically 0.0 or 0.1
95/// * `threshold` - Edge detection threshold for binarizing weights
96/// * `config_opt` - Optional custom optimization configuration (uses defaults if None)
97///
98/// # Returns
99///
100/// `OptimizationResult` containing:
101/// - `weight_matrix`: Continuous-valued d×d weight matrix W from optimization
102/// - `adjacency_matrix`: Binary d×d adjacency matrix (0/1 entries) after thresholding
103/// - `constraint_violation`: Final acyclicity constraint h(W) (should be < 1e-8)
104/// - `iterations`: Number of optimization iterations performed
105/// - `final_score`: Final objective value F(W) = loss + regularization
106///
107/// # Errors
108///
109/// Returns descriptive error messages for:
110/// - Invalid data dimensions (n < 1 or d < 1)
111/// - Data contains NaN or Inf values
112/// - Data norm too large (||X||_F > 1e10)
113/// - Dimension too high (d > 500)
114/// - Invalid hyperparameters (lambda ∉ [0, 1] or threshold < 0)
115/// - Optimization failure or non-convergence
116/// - Numerical issues during matrix exponential computation
117///
118/// # Algorithm Overview
119///
120/// 1. **Input Validation**: Check data quality and hyperparameter ranges
121/// 2. **Initialization**: Random weight matrix W₀ ~ Uniform[-0.5, 0.5]
122/// 3. **Optimization**: Augmented Lagrangian with L-BFGS inner solver
123/// 4. **Thresholding**: Extract binary adjacency via hard thresholding
124/// 5. **Validation**: Verify DAG structure in result
125/// 6. **Return**: Packaged OptimizationResult with both continuous and discrete outputs
126///
127/// # Example
128///
129/// ```rust,no_run
130/// use notears::learn_dag;
131/// use ndarray::Array2;
132///
133/// // Load or generate data (n samples, d variables)
134/// let data = Array2::zeros((100, 5));
135///
136/// // Learn DAG structure
137/// let result = learn_dag(&data, 0.1, 0.3, None)?;
138///
139/// // Inspect results
140/// println!("Discovered edges: {}", result.edges().len());
141/// println!("Acyclicity constraint: {:.6e}", result.constraint_violation);
142///
143/// # Ok::<(), Box<dyn std::error::Error>>(())
144/// ```
145///
146/// # Hyperparameter Guidance
147///
148/// - **lambda** (L₁ regularization): Larger λ → sparser graph. Default: 0.1
149/// - **threshold** (edge detection): Larger θ → fewer edges. Default: 0.3
150/// - **config_opt**: Use defaults for most problems. Customize only if needed:
151///   - `max_outer_iterations`: Increase if non-convergence (default: 100)
152///   - `constraint_tolerance`: Tighter tolerance for better acyclicity (default: 1e-8)
153///   - `penalty_rho_init`: Start value for penalty parameter (default: 1.0)
154///   - `edge_threshold`: Alternative to `threshold` parameter (see below)
155///
156/// # Note on Threshold Parameter
157///
158/// The `threshold` parameter in `learn_dag` overrides the `edge_threshold` field
159/// in the custom `OptimizationConfig` if both are provided.
160///
161/// # Performance Notes
162///
163/// - Complexity: O(d³ log d) per iteration due to matrix exponential
164/// - Typical runtime: d < 20 nodes: milliseconds, d = 50 nodes: seconds
165/// - Memory: O(d²) for weight matrices, O(n·d) for data
166/// - Scaling: Recommended for d ≤ 500 (matrix exponential stability)
167///
168/// # Data Quality Recommendations
169///
170/// Before calling `learn_dag`, preprocess data to:
171/// - **Center**: Subtract mean to zero per column
172/// - **Scale**: Divide by standard deviation per column
173/// - **Check**: Verify no NaN/Inf values present
174///
175/// This is NOT done automatically; use `standardize_data` if needed:
176/// ```rust,no_run
177/// use notears::{learn_dag, standardize_data, WeightMatrix};
178/// use ndarray::Array2;
179///
180/// // Example: load your data as WeightMatrix (n × d)
181/// let data: WeightMatrix = Array2::zeros((100, 5)); // n=100 samples, d=5 variables
182/// let standardized = standardize_data(&data)?;
183/// let result = learn_dag(&standardized, 0.1, 0.3, None)?;
184/// println!("Discovered edges: {}", result.edges().len());
185/// # Ok::<(), Box<dyn std::error::Error>>(())
186/// ```
187pub fn learn_dag(
188    data: &WeightMatrix,
189    lambda: f64,
190    threshold: f64,
191    config_opt: Option<OptimizationConfig>,
192) -> Result<OptimizationResult, Box<dyn std::error::Error>> {
193    use rand::Rng;
194
195    let (n, d) = data.dim();
196
197    // ========== STEP 1: Input Validation ==========
198
199    // Check data dimensions
200    if n == 0 || d == 0 {
201        return Err("Data matrix must have positive dimensions (n > 0 and d > 0)".into());
202    }
203
204    if n < d {
205        return Err(format!(
206            "Sample size (n={}) must be ≥ number of variables (d={}). \
207             Underdetermined regression problem.",
208            n, d
209        ).into());
210    }
211
212    if d > 500 {
213        return Err(format!(
214            "Dimension d={} exceeds recommended maximum (d ≤ 500). \
215             Matrix exponential computation becomes numerically unstable.",
216            d
217        ).into());
218    }
219
220    // Check for NaN/Inf in data
221    if data.iter().any(|x| !x.is_finite()) {
222        return Err(
223            "Data matrix contains NaN or Inf values. \
224             Please clean data before calling learn_dag."
225                .into(),
226        );
227    }
228
229    // Check data norm (scaling guard)
230    let data_norm = utils::frobenius_norm(data);
231    if data_norm > 1e10 {
232        return Err(format!(
233            "Data Frobenius norm ({:.2e}) exceeds safety threshold (1e10). \
234             Consider rescaling data.",
235            data_norm
236        ).into());
237    }
238
239    if data_norm < 1e-10 {
240        return Err("Data is essentially zero (Frobenius norm < 1e-10). Please provide non-trivial data.".into());
241    }
242
243    // ========== STEP 2: Hyperparameter Validation ==========
244
245    // Validate lambda (L1 regularization)
246    if lambda < 0.0 || lambda > 1.0 {
247        return Err(format!(
248            "Regularization lambda={} not in valid range [0.0, 1.0]. \
249             Typical values: 0.0 (no sparsity) or 0.1.",
250            lambda
251        ).into());
252    }
253
254    // Validate threshold (non-negative)
255    if threshold < 0.0 {
256        return Err(format!(
257            "Edge detection threshold={} is negative. Must be ≥ 0.0.",
258            threshold
259        ).into());
260    }
261
262    // ========== STEP 3: Configuration Setup ==========
263
264    // Use provided config or create defaults
265    let mut config = config_opt.unwrap_or_else(|| {
266        OptimizationConfig::new(
267            100,       // max_outer_iterations
268            50,        // max_lbfgs_iterations
269            10,        // lbfgs_memory
270            1e-8,      // constraint_tolerance
271            1.0,       // penalty_rho_init
272            0.25,      // progress_rate
273            0.3,       // edge_threshold (will be overridden below)
274        ).unwrap_or_else(|_| OptimizationConfig::default())
275    });
276
277    // Override edge_threshold with provided threshold parameter
278    config.edge_threshold = threshold;
279
280    let reg_config = RegularizationConfig::new(lambda, false)?;
281
282    // ========== STEP 4: Weight Matrix Initialization ==========
283
284    // Random initialization: W₀_ij ~ Uniform[-0.5, 0.5]
285    // This breaks symmetry and accelerates convergence compared to zero initialization
286    let mut rng = rand::thread_rng();
287    let w_init = ndarray::Array2::from_shape_fn((d, d), |_| {
288        rng.gen_range(-0.5..0.5)
289    });
290
291    // ========== STEP 5: Run Optimization ==========
292
293    let result = solve_ecp(&w_init, data, &reg_config, &config)
294        .map_err(|e| format!("Optimization failed: {}", e))?;
295
296    // ========== STEP 6: Extract Discrete Adjacency ==========
297
298    let adjacency = extract_adjacency(&result.weight_matrix, threshold)
299        .map_err(|e| format!("Failed to extract adjacency matrix: {}", e))?;
300
301    // ========== STEP 7: Validation Checks ==========
302
303    // Warn if result is not strictly acyclic
304    if result.constraint_violation > 1e-6 {
305        eprintln!(
306            "[WARNING] Acyclicity constraint h(W) = {:.6e} > 1e-6. \
307             Result may not strictly represent a DAG.",
308            result.constraint_violation
309        );
310    }
311
312    // Verify extracted adjacency is acyclic
313    if !is_acyclic_adjacency(&adjacency) {
314        eprintln!(
315            "[WARNING] Extracted adjacency matrix contains cycles. \
316             Consider increasing edge_threshold for sparser structure."
317        );
318    }
319
320    // ========== STEP 8: Return Result ==========
321
322    Ok(OptimizationResult {
323        weight_matrix: result.weight_matrix,
324        adjacency_matrix: adjacency,
325        constraint_violation: result.constraint_violation,
326        iterations: result.iterations,
327        final_score: result.final_score,
328    })
329}

336    #[test]
337    fn test_complete_pipeline() -> Result<(), Box<dyn std::error::Error>> {
338        // Generate synthetic data: linear DAG with 3 nodes
339        // True structure: 1 -> 2 -> 3
340        let n = 100;
341        let mut data = Array2::zeros((n, 3));
342
343        for i in 0..n {
344            data[[i, 0]] = (i as f64 % 10.0) / 10.0; // X1 ~ [0, 1]
345            data[[i, 1]] = data[[i, 0]] * 0.5 + ((i as f64 % 3.0) / 10.0 - 0.15); // X2 = 0.5*X1 + noise
346            data[[i, 2]] = data[[i, 1]] * 0.8 + ((i as f64 % 2.0) / 10.0 - 0.1); // X3 = 0.8*X2 + noise
347        }
348
349        // Standardize
350        let std_data = standardize_data(&data)?;
351
352        // Run solver
353        let config = OptimizationConfig::new(500, 50, 10, 1e-6, 1.0, 0.25, 0.3)?;
354        let loss_config = RegularizationConfig::new(0.1, false)?;
355
356        let result = solve_with_config(&std_data, config, loss_config)?;
357
358        // Check results
359        assert!(result.constraint_violation >= 0.0);
360        assert!(result.final_score > 0.0);
361
362        // Detect edges
363        let edges = result.edges();
364        println!(
365            "Learned {} edges. Acyclicity constraint: {:.6}",
366            edges.len(),
367            result.constraint_violation
368        );
369
370        Ok(())
371    }
